slot(name)

Marks this slot as eligible to make ad calls.

Params
name string 
    slot: function (name) {
        var slot = SlotFactory.get(name);
        if (!slot.mock) {
            slot.enabled = true;
        }
    },

group(name)

Marks each slot in this group as eligible to make ad calls.

Params
name string 
    group: function (name) {
        var group = GroupFactory.create(name);
        group.forEach(function (slot) {
            slot.enabled = true;
        });
    }
};
