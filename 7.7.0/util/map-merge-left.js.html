Map Merge Left

Shallow merge two Objects.

Params
first Object.<string, *>
second Object.<string, *>
Returns
Object.<string, *> 
module.exports = function (first, second) {
    var key;
    first = first || {};
    second = second || {};
    for (key in second) {
        first[key] = second[key];
    }
    return first;
};
