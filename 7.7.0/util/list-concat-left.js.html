List Concat Left

Concat arrays across two Objects for all keys.

Params
first Object.<string, function|Array.<function>>
second Object.<string, function|Array.<function>>
Returns
Object.<string, Array.<function>> 
module.exports = function (first, second) {
    var name;
    first = first || {};
    second = second || {};
    for (name in first) {
        first[name] = [].concat(first[name]);
    }
    for (name in second) {
        first[name] = [].concat(
            first[name] || [],
            second[name]
        );
    }
    return first;
};
