scrubConf(conf)

Ensures a slot's id is unique in the page. Ad containers have their ids mutated with the suffix -h#. To preserve an ad container's id, use the AdSlot setting preserveId.

Params
conf Object Configuration for a single ad slot.
Returns
Object Clean slot configuration. 
module.exports = function (conf) {
    var el = global.document.getElementById(conf.id),
        suffix = 0;

    if (el) {
        do {
            suffix += 1;
            conf.id = el.id + '-h' + suffix;
        } while (conf.id in cache);
        el.id = conf.id;
        cache[conf.id] = true;
        return conf;
    }
    throw Error('Ad slot ' + conf.name + ' container #' + conf.id + ' was not found!');
};
