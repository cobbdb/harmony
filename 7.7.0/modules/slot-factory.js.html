    create: function (conf) {
        var slot, set;
        try {
            if (!conf.preserveId) {
                conf = scrubConf(conf);
            }
            slot = Slot(conf);
            masterGroup.add(slot);
            if (slot.group) {
                set = [].concat(slot.group);
                set.forEach(function (groupName) {
                    var group = GroupFactory.create(groupName);
                    group.add(slot);
                });
            }
        } catch (err) {
            log('error', {
                conf: conf,
                msg: err.message
            });
        } finally {
            return slot || null;
        }
    },
